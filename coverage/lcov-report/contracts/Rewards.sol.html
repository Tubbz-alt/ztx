pragma solidity 0.4.24;
 
import "zeppelin-solidity/contracts/ownership/Ownable.sol";
import "zeppelin-solidity/contracts/math/SafeMath.sol";
import "./ZuluToken.sol";
import "./ZuluPassport.sol";
import "./VotableInterface.sol";
 
 
/**
 * @title Rewards contract - setting rules for the use of reserve tokens as rewards for Zulu Republic's citizens
 * @author Gustavo Guimaraes - <gustavo@zulurepublic.io>
 */
contract Rewards is Ownable {
    using SafeMath for uint256;
    address public zuluFoundation;
    uint256 public nextZuluFoundationPayment;
    uint256 public counterForFoundationPayout;
    bool public isCashBackEnabled;
 
    // external contract
    ZuluPassport public passport;
    VotableInterface public votable;
    ZuluToken public token;
 
    uint256 public tokensToRedeem;
    mapping (address => uint256) public redeemableTokens;
 
    event CashBackToggled(bool state);
    event CashBackRedeemed(address indexed citizen, uint256 value);
    event CashBackTriggered(address indexed citizen, uint256 value);
 
    /**
     * @dev Contract constructor function
     * @param _passport Zulu passport contract address
     * @param _votable VotableInterface contract address
     * @param _votable ZuluToken contract address
     * @param _zuluFoundation Address of the Zulu Foundation
     */
    constructor(
        address _passport,
        address _votable,
        address _token,
        address _zuluFoundation
    )
        public
    {
 
        Erequire(
            _passport != address(0) &&
            _votable != address(0) &&
            _zuluFoundation != address(0)
        );
 
        passport = ZuluPassport(_passport);
        votable = VotableInterface(_votable);
        token = ZuluToken(_token);
 
        isCashBackEnabled = true;
        zuluFoundation = _zuluFoundation;
    }
 
    /**
     * @dev Toggle cash back flag
     */
    function toggleCashBack() external onlyOwner {
        isCashBackEnabled = !isCashBackEnabled;
        emit CashBackToggled(isCashBackEnabled);
    }
 
    /**
     * @dev Mint more tokens to rewards address
     * @param _amount Amount of tokens to be minted for rewards contract
     */
    function mintTokensForRewardsContract(uint256 _amount) external {
        Erequire(votable != address(0) && _amount != 0);
 
        bytes32 hashMsg = keccak256(abi.encodePacked(
            _amount,
            keccak256(abi.encodePacked("numTokensToMintForRewardsContract"))
            ));
 
        require(votable.getHasBeenSuccessful(hashMsg));
 
        votable.implementSuccessfulVotingOnRewardsTokens(hashMsg);
        token.mint(address(this), _amount);
    }
 
    /**
     * @dev Pay percentage of reserve tokens to Zulu Foundation annually
     */
    function payoutFoundationFromReserve() external {
        require(now >= nextZuluFoundationPayment && counterForFoundationPayout < votable.maxYearsForFoundationPayout());
 
        uint256 tokensAvailable = token.balanceOf(this).sub(tokensToRedeem);
        uint256 tokensForFoundationFromReserve = tokensAvailable.mul(votable.foundationPayoutPercentage()).div(100);
 
        nextZuluFoundationPayment = now.add(365 days);
        counterForFoundationPayout = counterForFoundationPayout.add(1);
 
        token.transfer(zuluFoundation, tokensForFoundationFromReserve);
    }
 
    /**
     * @dev Redeem accrued cashback tokens
     */
    function redeemCashback() external returns(bool) {
        uint256 tokens = redeemableTokens[msg.sender];
        redeemableTokens[msg.sender] = 0;
 
        tokensToRedeem = tokensToRedeem.sub(tokens);
        emit CashBackRedeemed(msg.sender, tokens);
        return token.transfer(msg.sender, tokens);
    }
 
    /**
     * @dev Provide tokens for cashback. Only callable by owner
     * @param _from Citizen's address
     * @param _to Merchant's address
     * @param _value Amount exchanged between citizen and merchant
     */
    function giveOutCashBack(address _from, address _to, uint256 _value) external onlyOwner {
        uint256 cashbackValue = _value.mul(votable.cashBackPercentage()).div(100);
 
        if (
            isCashBackEnabled &&
            tokensToRedeem.add(cashbackValue) <= rewardsBalance() &&
            passport.isCitizen(_from) &&
            passport.isMerchant(_to)
        ) {
 
            cashback(_from, cashbackValue);
        }
    }
 
    function cashback(address beneficiary, uint256 _value) internal {
        redeemableTokens[beneficiary] = redeemableTokens[beneficiary].add(_value);
        tokensToRedeem = tokensToRedeem.add(_value);
 
        emit CashBackTriggered(beneficiary, _value);
    }
 
    function rewardsBalance() internal view returns(uint256) {
        return token.balanceOf(this).sub(tokensToRedeem);
    }
}

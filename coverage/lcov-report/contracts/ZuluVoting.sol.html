pragma solidity 0.4.24;
import "zeppelin-solidity/contracts/math/SafeMath.sol";
import "./ZuluPassport.sol";
 
 
/**
 * @title Zulu Voting contract
 * @author Timo Hedke - <timo@zulurepublic.io>
 */
contract ZuluVoting {
    using SafeMath for uint256;
 
    ZuluPassport public passport;
    uint256 public minNumOfCitizens;
 
    struct VotingStatus {
        bool initialized;
        bool hasBeenSuccessful;
        uint256 counter;
        bytes32 pollNonce;
    }
 
    // Count used to reset successful vote to make possible to vote again in future
    mapping (bytes32 => mapping(bytes32 => bool)) public voted; // Voted by Citizen
    mapping (bytes32 => VotingStatus) public votes;             // Votes of Citizen for initiative
 
    event Voted(address citizen, uint256 value, bool vote);
    event VoteSuccessful(uint256 value);
 
    /**
     * @dev Contract constructor function
     * @param _passport Zulu passport contract
     * @param _minNumOfCitizens Figure for the minimum number of citizens Zulu must have for voting
     */
    constructor(
        address _passport,
        uint256 _minNumOfCitizens
    )
        public
    {
        Erequire(_passport != address(0));
 
        passport = ZuluPassport(_passport);
        minNumOfCitizens = _minNumOfCitizens;
    }
 
    /**
     * @dev Vote on reserve value
     * @param _value Value to vote on
     * @param _vote If against or for it
     * @param _option Byte of proposal
     */
    function vote(uint256 _value, bool _vote, bytes32 _option) external {
        bytes32 hashMsg = keccak256(abi.encodePacked(_value, _option));
        bytes32 hashSender = keccak256(abi.encodePacked(msg.sender, hashMsg));
 
        if (!votes[hashMsg].initialized) {
            votes[hashMsg].initialized = true;
            votes[hashMsg].pollNonce = keccak256(abi.encodePacked("initalizePoll"));
        }
 
        checkEligibility(msg.sender, hashMsg, hashSender, _vote);
 
        citizenVotes(msg.sender, hashMsg, hashSender, _vote, _value);
 
        checkForSuccessfulVoting(hashMsg, _value);
    }
 
    /**
     * @dev Getter for checking the struct bool value
     * @param hashMsg Bytes for the message hash
     * @return bool whether vote on a proposal has been successful or not
     */
    function getHasBeenSuccessful(bytes32 hashMsg) public view returns(bool) {
        return votes[hashMsg].hasBeenSuccessful;
    }
 
    /**
     * @dev Mark as successful implementation of the voting by token contract
     * @param hashMsg Hash of the value and proposal
     */
    function voteSuccessfullyImplemented(bytes32 hashMsg) internal {
        Erequire(getHasBeenSuccessful(hashMsg));
 
        votes[hashMsg].hasBeenSuccessful = false;
        votes[hashMsg].counter = 0;
        //next poll can start
        votes[hashMsg].pollNonce = keccak256(abi.encodePacked(votes[hashMsg].pollNonce));
    }
 
    /**
     * @dev Check if vote is eligible
     * @param _citizen Address of msg sender
     * @param _hashMsg Hash of vote value and option
     * @param _hashSender Hash msg sender and hashMsg
     * @param _vote Bool if pro or con the topic
     */
    function checkEligibility
    (
        address _citizen,
        bytes32 _hashMsg,
        bytes32 _hashSender,
        bool _vote
    )
        internal
        view
    {
        Erequire(passport.isCitizen(_citizen));
        Erequire(minNumOfCitizens < passport.numOfCitizens());
        require(!votes[_hashMsg].hasBeenSuccessful);
        require(voted[votes[_hashMsg].pollNonce][_hashSender] != _vote);
    }
 
    /**
     * @dev Adds vote of _citizen to poll
     * @param _citizen Address of msg sender
     * @param _hashMsg Hash of vote value and option
     * @param _hashSender Hash msg sender and hashMsg
     * @param _vote Bool if pro or con the topic
     * @param _value Value to vote on
     */
    function citizenVotes
        (
            address _citizen,
            bytes32 _hashMsg,
            bytes32 _hashSender,
            bool _vote,
            uint256 _value
        )
            internal
    {
        voted[votes[_hashMsg].pollNonce][_hashSender] = _vote;
 
        if (_vote) {
            votes[_hashMsg].counter = votes[_hashMsg].counter.add(1);
        } else {
            votes[_hashMsg].counter = votes[_hashMsg].counter.sub(1);
        }
 
        emit Voted(_citizen, _value, _vote);
    }
 
    /**
     * @dev Checks if needed votes (50%) is reached
     * @param _hashMsg Hash of vote value and option
     * @param _value Value to vote on
     */
    function checkForSuccessfulVoting( bytes32 _hashMsg, uint256 _value) internal {
        if (votes[_hashMsg].counter > passport.numOfCitizens().mul(50).div(100)) {
            votes[_hashMsg].hasBeenSuccessful = true;
 
            emit VoteSuccessful(_value);
        }
    }
 
}

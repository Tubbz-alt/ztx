pragma solidity 0.4.24;
 
import "zeppelin-solidity/contracts/math/SafeMath.sol";
import "./upgradeable/OwnableUpgradeable.sol";
 
 
/**
 * @title ZuluPassport - members of the zulu republic
 * @author Gustavo Guimaraes - <gustavo@zulurepublic.io>
 */
contract ZuluPassport is OwnableUpgradeable {
    using SafeMath for uint256;
 
    struct Citizen {
        bool isCitizen;
        bytes32 passportData;
    }
 
    struct Merchant {
        bool isMerchant;
        bytes32 registryData;
    }
 
    uint256 public numOfCitizens;
    uint256 public numOfMerchants;
 
    mapping (address => Citizen) public citizen;
    mapping (address => Merchant) public store;
 
    event LogCitizen(address indexed citizenAddress, string action);
    event LogMerchant(address indexed merchantAddress, string action);
 
    modifier isNotContract(address addr) {
        uint256 size;
        assembly { size := extcodesize(addr) }
 
        require(size == 0);
        _;
    }
 
    /**
     * @dev Add citizen
     * @param citizenAddress Must be an externally owned account
     * @param passportData Encrypted data about the citizen
     */
    function addCitizen(address citizenAddress, bytes32 passportData)
        public
        onlyOwner
        isNotContract(citizenAddress)
    {
        require(
            citizenAddress != address(0) &&
            passportData != 0 &&
            !citizen[citizenAddress].isCitizen &&
            !store[citizenAddress].isMerchant
        );
 
        numOfCitizens = numOfCitizens.add(1);
 
        citizen[citizenAddress].passportData = passportData;
        citizen[citizenAddress].isCitizen = true;
        emit LogCitizen(citizenAddress, "add");
    }
 
    /**
     * @dev Update citizen passport
     * @param citizenAddress Address of citizen who is already in the ZuluPassport contract
     * @param newPassportData Update of the encrypted data about the citizen
     */
    function updateCitizen(address citizenAddress, bytes32 newPassportData) public onlyOwner {
        require(citizenAddress != address(0) && newPassportData != 0 && citizen[citizenAddress].isCitizen);
 
        citizen[citizenAddress].passportData = newPassportData;
        emit LogCitizen(citizenAddress, "update");
    }
 
    /**
     * @dev Remove citizen
     * @param citizenAddress Address of citizen who is already in the ZuluPassport contract
     */
    function removeCitizen(address citizenAddress) public onlyOwner {
        require(citizenAddress != address(0) && citizen[citizenAddress].isCitizen);
 
        numOfCitizens = numOfCitizens.sub(1);
        citizen[citizenAddress].isCitizen = false;
        emit LogCitizen(citizenAddress, "remove");
    }
 
    /**
     * @dev Convinience function for checking whether an eth address is citizen
     * @param citizenAddress Address to check
     * @return bool
     */
    function isCitizen(address citizenAddress) public view returns (bool) {
        return citizen[citizenAddress].isCitizen;
    }
 
    /**
     * @dev Add store
     * @param merchantAddress Must be an externally owned account
     * @param registryData Encrypted data about the merchant
     */
    function addMerchant(address merchantAddress, bytes32 registryData)
        public
        onlyOwner
        isNotContract(merchantAddress)
    {
        require(
            merchantAddress != address(0) &&
            registryData != 0 &&
            !store[merchantAddress].isMerchant &&
            !citizen[merchantAddress].isCitizen
        );
 
        numOfMerchants = numOfMerchants.add(1);
        store[merchantAddress].registryData = registryData;
        store[merchantAddress].isMerchant = true;
        emit LogMerchant(merchantAddress, "add");
    }
 
    /**
     * @dev Update store registry
     * @param merchantAddress Address of merchant
     * @param newRegistryData Update of the encrypted data about the merchant
     */
    function updateMerchant(address merchantAddress, bytes32 newRegistryData) public onlyOwner {
        require(merchantAddress != address(0) && newRegistryData != 0 && store[merchantAddress].isMerchant);
 
        store[merchantAddress].registryData = newRegistryData;
        emit LogMerchant(merchantAddress, "update");
    }
 
    /**
     * @dev Remove store
     * @param merchantAddress Address of merchant
     */
    function removeMerchant(address merchantAddress) public onlyOwner {
        require(merchantAddress != address(0) && store[merchantAddress].isMerchant);
 
        numOfMerchants = numOfMerchants.sub(1);
        store[merchantAddress].isMerchant = false;
        emit LogMerchant(merchantAddress, "remove");
    }
 
    /**
     * @dev Check whether eth address is merchant
     * @param merchantAddress Address to check
     * @return bool
     */
    function isMerchant(address merchantAddress) public view returns (bool) {
        return store[merchantAddress].isMerchant;
    }
}

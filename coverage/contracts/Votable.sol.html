pragma solidity 0.4.24;
 
import "./ZuluVoting.sol";
import "./VotableInterface.sol";
 
 
contract Votable is VotableInterface, ZuluVoting {
    uint256 public maxYearsForFoundationPayout_;
    uint256 public foundationPayoutPercentage_;
    uint256 public cashBackPercentage_;
    address public rewards;
    event NewValueSetByVoting(uint256 value, string action);
 
    /**
     * @dev Contract constructor function
     * @param _passport Zulu passport contract
     * @param _minNumOfCitizens Figure for the minimum number of citizens Zulu must have for voting
     */
    constructor(
        address _passport,
        uint256 _minNumOfCitizens
    )
        public
        ZuluVoting(_passport, _minNumOfCitizens)
    {
        maxYearsForFoundationPayout_ = 5; // 5 years initially
        cashBackPercentage_ = 5; // 5% initially
        foundationPayoutPercentage_ = 1; // 1% initially
    }
 
    modifier onlyRewardsContract() {
        require(msg.sender == rewards);
        _;
    }
 
    /**
     * @dev Set the reward contract
     * @param _rewards Address of reward contract
     */
    function setRewardsContract(address _rewards) external {
        Erequire(_rewards != address(0) && rewards == address(0)); // only allowed to be set once
        rewards = _rewards;
    }
 
    /**
     * @dev implement successful voting on rewards tokens
     * @param hashMsg Hash of the value with proposal
     */
    function implementSuccessfulVotingOnRewardsTokens(bytes32 hashMsg) external onlyRewardsContract {
        Erequire(rewards != address(0));
 
        voteSuccessfullyImplemented(hashMsg);
    }
 
    /**
     * @dev Set new maxYearsForFoundationPayout
     * @param _value Years
     */
    function setMaxYearsForFoundationPayout(uint256 _value) external {
        Erequire(_value != 0);
        bytes32 hashMsg = keccak256(abi.encodePacked(
            _value,
            keccak256(abi.encodePacked("maxYearsForFoundationPayout"))
            ));
 
        if (getHasBeenSuccessful(hashMsg)) {
            maxYearsForFoundationPayout_ = _value;
            emit NewValueSetByVoting(_value, "maxYearsForFoundationPayout");
            voteSuccessfullyImplemented(hashMsg);
        }
    }
 
    /**
     * @dev Set new foundationPayoutPercentage
     * @param _value Figure for var setting
     */
    function setFoundationPayoutPercentage(uint256 _value) external {
        Erequire(_value != 0);
        bytes32 hashMsg = keccak256(abi.encodePacked(
                _value,
                keccak256(abi.encodePacked("foundationPayoutPercentage"))
            ));
 
        if (getHasBeenSuccessful(hashMsg)) {
            foundationPayoutPercentage_ = _value;
            emit NewValueSetByVoting(_value, "foundationPayoutPercentage");
            voteSuccessfullyImplemented(hashMsg);
        }
    }
 
    /**
     * @dev Set new reserve cashback percentage value
     * @param _value Number to set cashback percentage
     */
    function setCashBackPercentage(uint256 _value) external {
        Erequire(_value >= 0 && _value <= 100, "Value needs to be between 0 and 100");
        bytes32 hashMsg = keccak256(abi.encodePacked(_value, keccak256(abi.encodePacked("cashBackPercentage"))));
 
        if (getHasBeenSuccessful(hashMsg)) {
            cashBackPercentage_ = _value;
            emit NewValueSetByVoting(_value, "cashBackPercentage");
            voteSuccessfullyImplemented(hashMsg);
        }
    }
 
    /**
     * @dev max number of years for foundation payout
     */
    function maxYearsForFoundationPayout() external view returns (uint256) {
        return maxYearsForFoundationPayout_;
    }
 
    /**
     * @dev number of foundation payout percentage
     */
    function foundationPayoutPercentage() external view returns (uint256) {
        return foundationPayoutPercentage_;
    }
 
    /**
     * @dev number of cashback percentage
     */
    function cashBackPercentage() external view returns (uint256) {
        return cashBackPercentage_;
    }
}
